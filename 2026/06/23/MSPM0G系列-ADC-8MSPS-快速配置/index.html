    概述

    本篇讲解如何快速配置 MSPM0G 系列 MCU 的 ADC 工作在 8MSPS 的采样率，核心原理是使用交错采样技术，配置两个 4MSPS 的 ADC，并使用反相的触发源驱动 ADC 采样。有关具体原理的讲解请移步至 MSPM0G系列 ADC-数据流讲解。
    进入正式配置前，读者需将 MCU 主频调整至 80MHz。不熟悉倍频步骤的同学，可移步至 MSPM0G系列 时钟树讲解。

    总共要配置两次 ADC，使用不同的 ADC 内核，将会占满 MSPM0G 系列的 ADC 资源。单个 ADC 的配置流程与 MSPM0G系列 ADC-4MSPS-快速配置 类似，没有尝试过 4MSPS 采样速率的同学可查看该快速配置文档进行练手。
    两次 ADC 配置有细微差别，体现在 DMA Configuration 和 Event Configuration 中。其中某一个 ADC 的详细配置流程如下：

    另一 ADC 的配置与前一 ADC 的配置类似，区别如下：

    • DMA Configuration/Enable DMA Triggers 设为 MEM9
    • DMA Configuration/DMA Channel/Transfer Size 设为 1110
    • Event Configuration 中需订阅2通道事件，也就是 down 事件

    现对以上配置做简易说明：

    1. PWM 设为 Center-aligned 模式。可获得边沿对称的 PWM 信号，从而严格驱动两个 ADC 以反相的形式进行交错采样。
    2. 是否设置定时器在上电后自动启动并不重要，因为由 sysconfig 生成的 ti_msp_dl_config.c 文件中，各外设初始化的顺序不满足 ADC 交错采样的时序要求，上电后在 DMA 目的地数组中存储的第一批数据总是错误的。
    3. 两个 ADC 工作速率相同，若两者采样长度相同，则必在相当接近的时刻产生 DMA 传输完成中断。而在 ADC 的中断回调函数中又必须以极快的速度、严格的顺序完成 关闭 ADC 转换 - 关闭 DMA 传输 - 关闭事件订阅 三个重要操作。在执行某一 ADC 的中断回调函数时，被另一 ADC 的中断事件打断，会对数据产生不可预估的影响。因此需要调整其中一个 ADC 的 DMA 传输大小偏大（或偏小），使得两个 ADC 中断错峰发生。取数据时以偏小的为边界即可。
    4. 由 MSPM0G 系列 MCU 的 DMA 设计特点，两个 ADC 的 DMA传输触发时机需错峰产生，因此将第二个 ADC 的 DMA 触发设为 MEM9 装载时刻。该部分涉及 DMA 与 ADC 数据流的深入讲解，可移步至 MSPM0G系列 DMA-详细讲解 和 MSPM0G系列 ADC-数据流讲解。

    /*
8MSPS ADC简易验证工程
uart为uart1，rx-PB7，tx-PB6，
波特率460800，校验位-even，数据位-8 bit，停止位-1 bit
*/
#include "ti_msp_dl_config.h"

#define ADC_0_SAMPLE_SIZE (2100)
#define ADC_1_SAMPLE_SIZE (2220)

volatile uint16_t ADCSamples_0[ADC_0_SAMPLE_SIZE]; 
volatile uint16_t ADCSamples_1[ADC_1_SAMPLE_SIZE];
volatile bool CheckADC_0;
volatile bool CheckADC_1;

void UART_transmit_for_vofa(uint16_t data1, uint16_t data2);// 请读者自行实现
void UART_send_str(const char *str);//请读者自行实现

int main(void)
{
    SYSCFG_DL_init();

    NVIC_ClearPendingIRQ(ADC12_0_INST_INT_IRQN);
    NVIC_EnableIRQ(ADC12_0_INST_INT_IRQN);
    NVIC_ClearPendingIRQ(ADC12_1_INST_INT_IRQN);
    NVIC_EnableIRQ(ADC12_1_INST_INT_IRQN);

    DL_DMA_setSrcAddr(DMA, DMA_CH0_CHAN_ID, (uint32_t) DL_ADC12_getFIFOAddress(ADC12_0_INST));
    DL_DMA_setSrcAddr(DMA, DMA_CH1_CHAN_ID, (uint32_t) DL_ADC12_getFIFOAddress(ADC12_1_INST));
    DL_DMA_setDestAddr(DMA, DMA_CH0_CHAN_ID, (uint32_t) &ADCSamples_0[0]);
    DL_DMA_setDestAddr(DMA, DMA_CH1_CHAN_ID, (uint32_t) &ADCSamples_1[0]);
    DL_DMA_enableChannel(DMA, DMA_CH0_CHAN_ID);
    DL_DMA_enableChannel(DMA, DMA_CH1_CHAN_ID);

    DL_Timer_startCounter(PWM_0_INST);//此处需读者根据具体配置情况自行决定
    while (1) {
        if (CheckADC_0 && CheckADC_1) {
            UART_send_str("\ngood\n");
            CheckADC_0 = false;
            CheckADC_1 = false;
            
            DL_Timer_stopCounter(PWM_0_INST); 

            DL_ADC12_enableConversions(ADC12_0_INST);
            DL_ADC12_enableConversions(ADC12_1_INST);
            DL_ADC12_setSubscriberChanID(ADC12_0_INST, ADC12_0_INST_SUB_CH);
            DL_ADC12_setSubscriberChanID(ADC12_1_INST, ADC12_1_INST_SUB_CH);
            DL_DMA_setDestAddr(DMA, DMA_CH0_CHAN_ID, (uint32_t) &ADCSamples_0[0]);
            DL_DMA_setDestAddr(DMA, DMA_CH1_CHAN_ID, (uint32_t) &ADCSamples_1[0]);
            DL_DMA_enableChannel(DMA, DMA_CH0_CHAN_ID);
            DL_DMA_enableChannel(DMA, DMA_CH1_CHAN_ID);
            
            DL_Timer_startCounter(PWM_0_INST);

            for (int i = 1; i < ADC_0_SAMPLE_SIZE; ++i) 
                UART_transmit_for_vofa(ADCSamples_0[i-1], ADCSamples_1[i]);
        }
    }
}
//------ 中断回调 ------
void ADC12_0_INST_IRQHandler(void) {
    switch (DL_ADC12_getPendingInterrupt(ADC12_0_INST)) {
        case DL_ADC12_IIDX_DMA_DONE:
            //此处顺序不可随意颠倒
            DL_ADC12_disableConversions(ADC12_0_INST);
            DL_DMA_disableChannel(DMA, DMA_CH0_CHAN_ID);
            DL_ADC12_setSubscriberChanID(ADC12_0_INST, 0);
            CheckADC_0 = true;
            break;
        default:
            break;
    }
}
void ADC12_1_INST_IRQHandler(void) {
    switch (DL_ADC12_getPendingInterrupt(ADC12_1_INST)) {
        case DL_ADC12_IIDX_DMA_DONE:
            //此处顺序不可随意颠倒
            DL_ADC12_disableConversions(ADC12_1_INST); 
            DL_DMA_disableChannel(DMA, DMA_CH1_CHAN_ID);
            DL_ADC12_setSubscriberChanID(ADC12_1_INST, 0);
            CheckADC_1 = true;
            break;
        default:
            break;
    }
}
